VERT
#version 410 core
layout (location=0) in vec3 aPos;

uniform mat4 uProjection;
uniform mat4 uView;
uniform mat4 uModel;

void main()
{
	gl_Position = uProjection * uView * uModel * vec4(aPos, 1.0);
}
###
FRAG
#version 410 core
out vec4 FragColor;

uniform vec3 uColor;

void main()
{
	FragColor = vec4(uColor, 1.0f);
}
###
